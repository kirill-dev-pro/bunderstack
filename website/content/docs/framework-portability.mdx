# Framework Portability

`app.handler(req: Request): Promise<Response>` — every modern TS framework knows this shape.

## Standalone Bun

```ts
// server.ts
import { app } from './bunderstack'
Bun.serve({ port: 3001, fetch: app.handler })
```

## Next.js (App Router)

```ts
// app/api/[...bunderstack]/route.ts
import { getApp } from '@/bunderstack'

export async function GET(req: Request) {
  return (await getApp()).handler(req)
}
export const POST = GET
export const PATCH = GET
export const DELETE = GET
```

```ts
// bunderstack.ts — lazy singleton for Next.js
import { createBunderstackAsync } from 'bunderstack'
import * as schema from './schema'

let _app: Awaited<ReturnType<typeof createBunderstackAsync<typeof schema>>>

export async function getApp() {
  return (_app ??= await createBunderstackAsync({ schema, auth: { emailAndPassword: { enabled: true } } }))
}
```

## TanStack Start

```ts
// src/routes/api/$.tsx
import { createFileRoute } from '@tanstack/react-router'
import { app } from '~/bunderstack'

const handle = ({ request }: { request: Request }) => app.handler(request)

export const Route = createFileRoute('/api/$')({
  server: {
    handlers: {
      GET: handle,
      POST: handle,
      PATCH: handle,
      DELETE: handle,
    },
  },
})
```

TanStack Start supports top-level `await`, so `bunderstack.ts` can export `app` directly:

```ts
// src/bunderstack.ts
import { createBunderstackAsync } from 'bunderstack'
import * as schema from './schema'
import { access } from './access'

export const app = await createBunderstackAsync({ schema, access, auth: { emailAndPassword: { enabled: true } } })
```
